# building project:
cd ~
rm cpp-lib -r
mkdir cpp-lib
cp /mnt/c/code/cpp-lib/{includes,impl,testing} ./cpp-lib -r
cd cpp-lib
g++ -o testing/testing.elf -g -std=c++2a **/*.cpp -lpthread

# testing project:
cd ~/cpp-lib/testing
./testing.elf out img regexglob

# testing project with valgrind:
cd ~/cpp-lib/testing
valgrind --leak-check=full ./testing.elf out img regexglob
